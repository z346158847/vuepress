拷贝文件及文件夹并覆盖
```
@echo off
echo 开始更新博客。。。请稍后
xcopy /y /c /h /r /e "D:\workspace\webstorm\wjzhang\vuepress\vuepress\public\*.*" "D:\workspace\webstorm\wjzhang\ss\"
echo 博客更新完成
echo. & pause
```
获取系统时间
提交git
```
@echo off
echo 开始更新博客。。。请稍后
set d=%date:~0,10%
set t=%time:~0,8%
echo %d% %t%
git add .
git commit -m "%d% %t%"
git push origin master
echo 博客更新完成
echo. & pause
```
